import {PluginStore} from '@synchrocode/core';
import {EvidenceUploader} from './evidence-uploader';

export class TaskDetailRenderer {
private taskId: string;
private uploader: EvidenceUploader;

constructor(id: string) {
this.taskId = id;
this.uploader = new EvidenceUploader(id);
}

public async render(): Promise<HTMLElement> {
const data = await PluginStore.getTaskById(this.taskId);
// Render description, history, evidence tabs
return createDetailTemplate(data);
}

public async uploadEvidence(file: File): Promise<void> {
await this.uploader.upload(file);
}
}
