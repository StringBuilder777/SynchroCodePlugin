import { SynchroCode } from '@synchro/core';
import { PluginOptions } from './types';
/**
* Initialize the SynchroCode connection
*/
async function initializeSync(options: PluginOptions) {
  try {
    const client = await SynchroCode.connect({
      apiKey: options.token,
      realtime: true
    });
    console.log('Connected to SynchroCode server');
    return client;
  } catch (err) {
    console.error('Sync failed:', err);
  }
}
export default initializeSync;
