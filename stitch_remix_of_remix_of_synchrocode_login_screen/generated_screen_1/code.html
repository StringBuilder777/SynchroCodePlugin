import{JWTPayload}from'@types/jwt';
import{ContainerRequestContext}from'quarkus-core';
 
// CU-036 - JWT Middleware
@Provider
public class JwtMiddleware implements ContainerRequestFilter{
  @Override
  public void filter(ContainerRequestContext ctx) {
    final String token=ctx.getHeaderString("Authorization");
    if(token==null) {
      ctx.abortWith(Response.status(401).build());
    }
  }
}
